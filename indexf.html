function fmtMoney(v){return new Intl.NumberFormat('en-US',{style:'currency',currency:'USD',maximumFractionDigits:0}).format(v||0)}

function fmtScore(v){ const n = Number(v); return n > 0 ? `+${n}` : String(n); }

function fmtPlainNumber(v){return new Intl.NumberFormat('en-US',{maximumFractionDigits:0}).format(v||0)}

function parseMoneyInput(value){

  const cleaned = String(value).replace(/[^0-9.-]/g,'');

  const num = Number(cleaned);

  return Number.isFinite(num) ? num : 0;

}

function cssSafe(t){return t.replace(/[^a-zA-Z0-9_-]/g,'_')}

function normalizeName(t){return t.toLowerCase().replace(/[.,]/g,' ').replace(/\(.*?\)/g,' ').replace(/\s+/g,' ').trim()}

function resolvePlayerName(raw){

  const n = normalizeName(raw);

  if(playerAliases[n]) return playerAliases[n];

  const directPlayerMatch = players.find(p => normalizeName(p) === n);

  if(directPlayerMatch) return directPlayerMatch;

  for(const [alias,canonical] of Object.entries(playerAliases)){ if(n.includes(alias)) return canonical; }

  return null;

}

function getEnteredPlaces(){

function computeParticipantTotals(){

  const places = getEnteredPlaces();

  const payouts = getCurrentPayouts();

  return participants.map(p=>{

    const total = p.golfers.reduce((sum,golfer)=>{

      const place = places[golfer];

      if(!place || !payouts[place]) return sum;

      return sum + payouts[place];

    },0);

    return {...p,total};

  }).sort((a,b)=>{

    if(b.total !== a.total) return b.total - a.total;

    return a.name.localeCompare(b.name);

  });

}

function sortPlayersForDisplay(){

  const places = getEnteredPlaces();

  return [...players].sort((a,b)=>{

    const pa = places[a];

    const pb = places[b];

    const aBlank = !(pa > 0);

    const bBlank = !(pb > 0);

    if (aBlank && bBlank) return a.localeCompare(b);

    if (aBlank) return 1;

    if (bBlank) return -1;

    if (pa !== pb) return pa - pb;

    return a.localeCompare(b);

  });

}

function parseLeaderboardText(text){

  const lines = String(text).split(/\r?\n/).map(s=>s.trim()).filter(Boolean);

  const foundMap = {};

  const unmatched = [];

  let updatedStamp = '';

  lines.forEach(line=>{

    if(line.startsWith('#') || line.startsWith('//')) return;

    const updatedMatch = line.match(/^updated\s*:\s*(.+)$/i);

    if(updatedMatch){

      updatedStamp = `Updated: ${updatedMatch[1].trim()}`;

      return;

    }

    if(/^[-–—]+$/.test(line)) return;


    let place = null;

    let rawName = '';


    // Format 1: "T12 Scheffler" or "12 Scottie Scheffler"

    let match = line.match(/^T?(\d{1,3})\s+(.+)$/i);

    if(match){

      place = Number(match[1]);

      rawName = match[2];

    } else {

      // Format 2: "Scheffler = T12", "Scheffler | 12", "Scheffler, 12", "Scheffler: T12"

      match = line.match(/^(.+?)\s*(?:=|\||,|:)\s*T?(\d{1,3})\s*$/i);

      if(match){

        rawName = match[1];

        place = Number(match[2]);

      }

    }


    if(!match){

      // Allow blank template lines like "Scheffler =" without reporting them as errors.

      if(/^.+?(?:=|\||,|:)\s*$/.test(line)) return;

      unmatched.push(line);

      return;

    }


    let cleanedName = rawName

      .replace(/^[\-–—:]+/,'')

      .replace(/\s+E$/i,'')

      .replace(/\s+[+-]\d+$/i,'')

      .trim();

    const resolved = resolvePlayerName(cleanedName);

    if(resolved && players.includes(resolved) && place > 0){

      foundMap[resolved] = Math.trunc(place);

    } else {

      unmatched.push(line);

    }

  });

  return {foundMap, unmatched, updatedStamp};

}


function applyPlaces(foundMap){

  players.forEach(player=>{

    const el = document.getElementById(`place-${cssSafe(player)}`);

    if(!el) return;

    el.classList.remove('loaded');

    if(Object.prototype.hasOwnProperty.call(foundMap, player)){

      el.value = String(foundMap[player]);

      el.classList.add('loaded');

    } else {

      el.value = '';

    }

  });

  renderInputs();

}


async function loadLeaderboardFile(){

  const status = getStatusEl();

  status.textContent = 'Loading leaderboard.txt...';

  try{

    const response = await fetch('leaderboard.txt?ts=' + Date.now(), {cache:'no-store'});

    if(!response.ok) throw new Error(`HTTP ${response.status}`);

    const text = await response.text();

    const {foundMap, unmatched, updatedStamp} = parseLeaderboardText(text);

    setUpdatedStamp(updatedStamp);

    applyPlaces(foundMap);

    renderParticipants();

    const filled = Object.keys(foundMap).length;

    let msg = `Loaded leaderboard.txt and applied ${filled} golfer place${filled===1?'':'s'}.`;

    if(unmatched.length) msg += ` ${unmatched.length} line${unmatched.length===1?' was':'s were'} ignored.`;

    status.textContent = msg;

  }catch(err){

    setUpdatedStamp('');

    status.textContent = 'Could not load leaderboard.txt. Make sure it is uploaded to the same GitHub Pages folder as index.html.';
